ctrl + alt + t - открытие терминала

ls - показывает содержимое папки
ls -a показывает сркытие файлы
ls -la показывает допольнительную информацию
ls -R показывает содержимое всех вложенных файлов папок в директории

cd - перемещение по директориям 
cd .. - перемещние на уровень выше 
cd перемещние в корневую директорию 


mkdir - создать директорию 
mkdir -p makers/week1/python/test создает весь путь без ошибки 

nano - редактор в терминале 

rmdir - удаляет только пустую папкку (выйдет если ошибка если папка не пустая)
rm -rf - удаляет файлы и папки 
rm * удаляет все (не делайте так)

zip  название архива название папки которую вы хотите архивировать- архивирует папку
zip dir.zip dir

grep - вывод строк по условию 


mv - перемещает или переименовывает 
mv название файла путь куда вы хотите перенести 

mv test.txt /path/to/directory - перемещение 

mv test.txt test1.txt - переимновать

head tail - выводит текст с начала или с конца

head -n 5 file_name
tail -n 1 file_name 


cat - выводит содержимое файла в терминал 
cat file > file перезапись
cat file >> file дозапись 


cp - копирование 

cp file_name/dir_name path/to/direcotry

echo - записать строки в файл через терминал

echo 'какая то строка' > file_name
echo 'какая то строка' >> file_name
